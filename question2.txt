package zoho;
import java.util.*;
public class test2 {

	public static void main(String[] args) {
		
		Scanner in=new Scanner(System.in);
		String input=in.nextLine();
		
		//split the words and assign the array name of (arr)
		
		String arr[]=input.split(" ");
		
		///this loop for traverse the word
		
		for(int i=0;i<arr.length;i++)
		{
			String ans="";
			
			//this loop for reverse the character
			
			for(int j=arr[i].length()-1;j>=0;j--)
			{
				ans+=arr[i].charAt(j);
			}
			//print the output
			System.out.print(ans+" ");
		}

	}

}
